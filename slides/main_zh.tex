\section{背景与问题}
\begin{frame}{研究背景：为什么要预测损失曲线？}
\textbf{大语言模型预训练极其昂贵。} 研究者希望在烧算力\emph{之前}就能比较不同学习率调度的效果，从而避免昂贵的试错。
\vfill
\textbf{核心问题——调度感知的损失曲线预测：}
\begin{center}
\hl{给定学习率调度 $\{\eta_t\}$，在训练开始前预测整条损失曲线 $L(t)$。}
\end{center}
\vfill
\textbf{决定性的检验标准：跨调度泛化。}
\begin{itemize}
\item 仅在 cosine 上拟合模型，测试其能否预测从未见过的 WSD（warmup-stable-decay）族曲线。
\item 这是区分"曲线拟合"与"真正理解调度动力学"的试金石。
\end{itemize}
\vfill
\textbf{当前最优方法。} Multi-Power Law (MPL, Luo et al. 2025) 及其匹配理论 Functional Scaling Law (FSL, Li et al. 2025)。
\end{frame}

\begin{frame}{基线方法：Multi-Power Law (MPL)}
\textbf{MPL 公式}（7 个参数）：
\[
L(t)=L_0+A\,S(t)^{-\alpha}+B\!\sum_{k\le t}\!\underbrace{(\eta_{k-1}-\eta_k)}_{\Delta\eta_k}\,\Big[1-\big(1+C\,\eta_k^{-\gamma}(S(t)-S(k))\big)^{-\beta}\Big]
\]
其中 $S(t)=\sum_{i\le t}\eta_i$ 为\hl{累积学习率}（ intrinsic time / 自然时间）。
\vfill
\textbf{各项含义：}
\begin{itemize}
\item $L_0 + A\,S^{-\alpha}$：主干幂律下降（偏置衰减）。
\item $B\cdot\mathrm{LD}(t)$：退火奖励——每次 LR 下降带来的额外 loss reduction 的累积。
\item $\eta_k^{-\gamma}$：MPL 唯一的"变化率"旋钮，凭经验拟合得到 $\gamma\approx 0.64$。
\end{itemize}
\vfill
\textbf{关键观察（本工作的出发点）。} MPL \emph{确实}使用了 LR 下降量 $\Delta\eta_k$，且有一个变化率因子 $\eta_k^{-\gamma}$，\hl{但对"变化多快"只有一个经验指数}。它本质预测的是\hl{准静态（绝热）loss}——假设损失始终 instantaneous 地处于当前 LR 对应的平衡值。
\end{frame}

\begin{frame}{复现基线：忠实重现 MPL 官方结果}
\textbf{数据来源。} MPL 官方公开损失曲线：Llama-2 风格解码器，$25/100/400$M 三个尺度，各 9 种调度（cosine, constant, WSD, WSDLD, WSDCon-3/9/18, cosine-72k, constant-72k）。
\vfill
\textbf{实验设置（遵循课程要求）。}
\begin{itemize}
\item \hl{训练集}：cosine\_24000.csv、cosine\_72000.csv（仅 cosine 调度）
\item \hl{测试集}：wsd\_20000\_24000、wsdld\_20000\_24000、wsdcon\_3/9/18（完全外推）
\item 目标函数：$\log L$ 残差上的 Huber 损失；优化器：L-BFGS-B，16 个随机起点取最优。
\end{itemize}

\section{核心发现}
\begin{frame}{发现：绝热律在快速衰减上留下系统性正残差}
\textbf{方法。} 用上述拟合良好的 MPL（全过程数据训练）预测留出曲线，观察预测残差 $r(t)=L_{\rm true}(t)-L_{\rm MPL}(t)$。
\vfill
\centering
\includegraphics[width=0.82\linewidth]{figs/fig_residual_fit.png}
\vfill
\raggedright
\textbf{残差的清晰特征：}
\begin{enumerate}
\item 稳定高 LR 段：残差 $\approx 0$（系统处于平衡）。
\item 第 20000 步 LR \hl{陡降}处：残差突然\hl{跳为正}。
\item 衰减越陡，残差越大：WSD/WSDLD 末段 $+(6\text{--}15)\times10^{-3}$，而 gradual cosine $\lesssim 4\times 10^{-3}$。
\item \hl{关键判据：} 一个 gradual cosine 和一个 sharp WSD 到达\emph{同一}最终 LR（$3\times 10^{-5}$），\hl{只有陡降的产生滞后}——同终点、变化越快滞后越大，正是非绝热（变化率依赖）效应的教科书特征。
\end{enumerate}
\end{frame}

\begin{frame}{物理解释：绝热极限 vs 非绝热滞后}
\begin{columns}
\begin{column}{0.56\textwidth}
\textbf{绝热图像。}
\begin{itemize}
\item 固定 LR $\eta$ 时，损失会向一个准静态值 $L_{\rm eq}(\eta)$ 弛豫。
\item cosine 衰减很缓慢 → 损失始终紧跟平衡值。
\item \hl{在 cosine 上拟合的 MPL，学到的正是 $L_{\rm eq}$。}
\end{itemize}
\vspace{0.6em}

\textbf{非绝热图像。}
\begin{itemize}
\item WSD 衰减\emph{太快}，损失\hl{来不及弛豫}到新的更低平衡值。
\item 系统滞留在旧的更高损失状态 → 正残差。
\item 当 LR 扫描速率 $|\dot{\eta}|$ 超过弛豫速率 $1/\tau$ 时，可观测量系统性落后于平衡值。
\end{itemize}
\end{column}
\begin{column}{0.40\textwidth}
\begin{block}{绝热律预测}
\centering
$L_{\rm pred} = L_{\rm eq}(\eta_t)$\\[4pt]
\small 只依赖当前 LR 和累计进度
\end{block}
\vspace{0.6em}
\begin{block}{真实损失}
\centering
$L_{\rm true} = L_{\rm eq}(\eta_t) + \underbrace{\Delta L_{\rm lag}}_{>0}$\\[4pt]
\small 额外依赖 LR \emph{变化率}
\end{block}
\vspace{0.6em}
\begin{alertblock}{核心问题}
\centering
$\Delta L_{\rm lag}$ 的结构是什么？能否从动力学\emph{推导}出来？
\end{alertblock}
\end{column}
\end{columns}
\end{frame}

\section{理论推导}
\begin{frame}{推导：从 AdamW 预条件动力学出发}
\textbf{模型设置。} 在 Hessian 特征基下展开，模 $i$ 具有：曲率 $\lambda_i$，梯度噪声标准差 $s_i$。
\vfill
\textbf{关键简化。} 当 $\beta_2\to 1$（实际训练中的典型值 0.95--0.999），Adam 的二阶矩预条件子收敛到 $v_i^*\approx \mathbb{E}[g_i^2] = \lambda_i^2 V_i + s_i^2$。在噪声主导的"山方向"，$v_i^*\approx s_i^2$ 为常数 → \hl{Adam 退化为预条件步长 $\eta_{\rm eff}=\eta/s_i$ 的 SGD}。
\vfill
\textbf{线性递推。} 二阶矩偏差 $\Delta_i = V_i - V_i^*(\eta_t)$ 满足精确线性递推。对慢模（$\eta\lambda_i/s_i \ll 1$，事后验证成立），$(1-\eta\lambda_i/s_i)^2\approx e^{-2\eta\lambda_i/s_i}$，解得：
\[
\Delta L(t)=\sum_i \underbrace{w_i}_{dV_i^*/d\eta}\Big[\underbrace{e^{-2(\lambda_i/s_i)(S-S')}}_{\text{S-time 指数核}}\circledast\underbrace{\mathrm{drop}}_{(\eta-\eta')_+}\Big](t)
\]
\vfill
\textbf{精确恒等式。} 权重之和 $=\sum_i w_i = \sum_i \frac{dV_i^*}{d\eta}\cdot\frac{\lambda_i}{2} = \textcolor{blue!60!black}{\frac{dL_{\rm eq}}{d\eta}}$。
\end{frame}

\begin{frame}{可用修正律：DropRelaxS}
将谱混合塌缩为最低容量的单极闭合（避免过拟合）：
\vfill
\begin{center}
\fbox{\parbox{0.92\textwidth}{\centering
\vspace{0.3em}
$\displaystyle L(t)=L_{\rm MPL}(t)+c\,\chi\sum_{k<t}e^{-\lambda_{\rm slow}(S_t-S_k)}(\eta_k-\eta_{k+1})_+\quad(\chi=dL_{\rm eq}/d\eta)$\\[0.3em]
\vspace{0.3em}
}}
\end{center}
\vfill
\textbf{各项含义：}
\begin{itemize}
\item $\chi = dL_{\rm eq}/d\eta$：平衡地板对 LR 的敏感度，\hl{独立可测}（见后）。
\item $\lambda_{\rm slow}$：有效预条件曲率，控制弛豫速率；在公开同架构尺度上\hl{经验稳定}（见后）。
\item $c\le 1$：将谱混合压缩为单极的校准因子，\hl{近似可迁移}。
\item 记忆变量是累积学习率 $S_t-S_k$，而非步数 $t-k$。
\end{itemize}
\vfill
\textbf{推出的六条可证伪预言：}
\begin{enumerate}\small
\item 符号为正；\item 因果依赖 LR decrements；\item 记忆在 S-time；\item 幅度 $=dL_{\rm eq}/d\eta$；\item noise-dominated 模：$\tau\propto 1/\eta$；\item 谱混合：双指数优于单指数。
\end{enumerate}
\end{frame}

\section{验证}
\begin{frame}{验证 1：核心预言 $\tau\propto 1/\eta$ 成立}
\textbf{实验设计。} 利用两段式 wsdcon 曲线（第 8000 步阶跃降到常数 LR），第二阶段 LR 分别取 $\eta\in\{3,9,18\}\times 10^{-5}$。拟合残差瞬态 $r(t)\propto e^{-(t-8000)/\tau}$ 提取弛豫时间 $\tau$。
\vfill
\begin{columns}
\begin{column}{0.46\textwidth}
\includegraphics[width=\linewidth]{figs/fig_tau_scaling.png}
\end{column}
\begin{column}{0.50\textwidth}
\textbf{结果：}
\begin{itemize}
\item 合并 $100$M+$400$M：\hl{$p=1.00\pm 0.18$}
\item 各尺度单独：$-1.06/-0.94/-0.51$（$100/400/25$M）
\item 受控 AdamW 模拟复现：\hl{$p=0.971$}，$R^2=0.9996$
\item $\beta_2\in[0.9,0.9999]$ 扫描：$\tau$ 平坦，CV 仅 1.1\%
\end{itemize}
\vspace{0.3em}
\textbf{推论：} $p=1$ 正是\hl{噪声主导}的理论值 → \emph{自洽地}验证了推导中 $v_i^*\approx s_i^2$ 的关键简化步骤。
\end{column}
\end{columns}
\vfill
\centering
\hl{经典弛豫时间 $\tau\propto 1/\eta$ 首次在 LLM 预训练的损失曲线残差上被实测确认。}
\end{frame}

\begin{frame}{验证 2：残差就是推出的项；谱混合确认}
\textbf{方法。} 将拟合良好 MPL 的残差对 $\Drop_{\lambda_{\rm slow}=10}$ 做\hl{过原点回归}（只有一个斜率参数，防止过拟合）。
\vfill
\textbf{结果。}
\begin{itemize}
\item $R^2 = 0.40/0.80/0.87$（$25/100/400$M），信号随模型规模增长。
\item 双指数核替代单指数：$R^2$ 再提升 $0.06\text{--}0.07$，\hl{直接证实谱混合预言}。
\item 公平对比：S-time 指数核胜 step-time 核（留出 MAE $-21\%$），胜无记忆地板（$-3\%$）。
\end{itemize}
\vfill
\textbf{关键控制。} 在\emph{单条}衰减曲线上，有限-$\lambda_{\rm slow}$ 核与绝热极限（$\lambda_{\rm slow}\to 0$）数值简并（corr$\approx 0.99$）。\hl{速率常数只能由跨调度的对比来约束}——这正是我们使用 wsdcon 系列（不同第二阶段 LR）的原因。
\end{frame}

\begin{frame}{验证 3：幅度可独立预测}
\textbf{方法。} 不依赖 DropRelaxS 的拟合，\hl{独立地}从两段式终值损失 vs 第二阶段 LR 的关系测 $dL_{\rm eq}/d\eta$（noise-floor 探针）。
\vfill
\textbf{结果。}
\begin{itemize}
\item 用独立探针预测的理论幅度 $\kappa_{\rm pred} = c\cdot dL_{\rm eq}/d\eta$。
\item 与 DropRelaxS 回归拟合的幅度 $\kappa_{\rm fit}$ 比较：
\[
\frac{\kappa_{\rm fit}}{\kappa_{\rm pred}} = 0.43\, /\, 0.51\, /\, 0.60 \quad (25/100/400\text{M})
\]
\item 比值跨尺度 CV 仅 \hl{14\%} → $c\approx 0.5$ 在公开同架构尺度上近似可迁移。
\item $<1$ 的物理原因：谱混合中快模贡献被单极压缩所忽略。
\end{itemize}
\vfill
\begin{block}{至此}
$\lambda_{\rm slow}$（经验稳定）+ $c$（留一校准）+ $dL_{\rm eq}/d\eta$（独立测量）= \hl{低校准跨尺度迁移。}
\end{block}
\end{frame}

\item $\lambda_{\rm slow}=10$（固定常数）
\item $c$ 取\emph{其他两个尺度}的留一法平均
\item $dL_{\rm eq}/d\eta$ 从目标自身的两段式探针测量（与测试曲线不同）
\end{itemize}

\begin{frame}{升级 1：B-恒等式——幅度本来就在 MPL 里}
\textbf{命题。} 在固定 $S$ 下比较"保持 $\eta$"与"降到 $\eta-d\eta$ 并等饱和"：MPL 退火项变化 $-B\,d\eta\,G(\cdot)\to -B\,d\eta$（$G\to1$）。
\[
\left.\partial_\eta L^{\rm MPL}_{\rm eq}\right|_S = B
\quad\Rightarrow\quad \text{MPL 自带的 } B \text{ 就是修正所需的地板敏感度 } dL_{\rm eq}/d\eta .
\]
\vfill
\textbf{验证（可证伪）。} $B=363.8/437.9/523.4$ vs 独立探针测量 $381.7/499.3/561.5$（偏差 $5$--$15\%$，三个尺度全部成立）。
\vfill
\begin{block}{零额外测量迁移链}
$\kappa = c''\,\eta_{\rm peak} B$（$c''$ 留一尺度平均，CV 14.7\%）→ 同协议 \hl{$-43\%$，6/6}，与探针链 $-44\%$ 持平，但\hl{不需要目标尺度的任何额外训练}。光滑-only 重拟合 $B$（漂移 $\le30\%$）仍保留 $-36\%\sim-40\%$。
\end{block}
\end{frame}

\begin{frame}{升级 2：二阶幅度闭合 $\chi(\eta)\propto\eta^{\delta}$}
\textbf{发现。} 平衡噪声地板\hl{超线性}：$F_{\rm eq}\propto\eta^{1+\delta}$，$1{+}\delta = 1.06/1.49/1.25$（25/100/400M 探针实测）。原闭合把 $\chi$ 冻结为常数，丢掉了推导中本来就有的 $\eta_k$ 依赖。
\[
L(t)=L_{\rm MPL}(t)+\kappa\sum_{k<t}\Big(\tfrac{\eta_k}{\eta_{\rm peak}}\Big)^{\!\delta}
e^{-\lambda_{\rm slow}(S_t-S_k)}\tfrac{(\eta_{k-1}-\eta_k)_+}{\eta_{\rm peak}}
\]
\vfill
\begin{itemize}
\item 公开曲线：$\delta{=}1/4$（Pareto 选取，从不在目标上拟合）令审计迁移矩阵\hl{所有聚合指标占优}；纯探针校准 $-17\%\to-23\%\sim-29\%$。
\item \hl{真实 10.7M 模型、从未用于选 $\delta$ 的界外调度}：sharp600 $-18\%\to-71\%$（$\delta{=}3/4$）；第二套 showcase 床 wsd\_sharp $-14\%\to-52\%$。
\item 诚实拆分：部分增益来自幅度再校准（$\le1$ 权重抬高拟合 $\kappa$）；oracle 等幅度下族内形状仍以 $\delta{=}0$ 略优。
\item 机制：NQM 仿射路线在量级上\hl{被证伪}（差 3 个量级）；bulk+edge 谱（追踪截断 $a_{\max}\!\propto\!1/\eta$）给出 $F=c_1\eta+c_2\eta^{\zeta}$——一致但边缘重适应步仍是 conjecture。
\item 等-$S$ 阶梯（消除骨干混淆）：10.7M 的地板指数 $p=0.73\,[0.62,0.86]$ \hl{亚线性}——指数随尺度涌现（$0.73\to1.06\to1.49\to1.25$），10M 上 $\delta>0$ 的增益来自浓度依赖的可见幅度而非地板曲率；$\delta$ \hl{无跨尺度普适值}，按尺度实测是唯一可辩护路线。
\end{itemize}
\end{frame}

\section{独立复现}
\begin{frame}{独立复现 1：真实 $\sim$10M transformer}
\textbf{动机。} 作者原始数据不再公开（仓库 404）。我们在\hl{不同硬件、不同模型、不同数据}上完全独立地重新训练模型，验证机制是否普适。
\vfill
\textbf{设置。} $\sim$10M 参数 Llama-2 风格 byte-level 解码器（RMSNorm + RoPE + SwiGLU），在 WikiText 上训练。AdamW：$\beta_1=0.9$，$\beta_2=0.95$，$\mathrm{wd}=0.1$。
\vfill
\textbf{三调度设计。} 到达\emph{同一}末端 LR（$2.5\times 10^{-3}\to 2.5\times 10^{-4}$），但扫到那里的速度不同：

\begin{frame}{独立复现 2：从零 AdamW 模拟}
\textbf{设置。} 噪声二次模型（NQM）：真实 AdamW 迭代（真实 $m,v$ EMA，真实 $m/(\sqrt{v}+\epsilon)$ 预条件器，真实高斯梯度噪声），Monte-Carlo 系综平均作为 ground truth。闭式线性理论\emph{仅}作独立基线，绝不既用来生成又用来拟合。
\vfill
\begin{columns}
\begin{column}{0.48\textwidth}
\textbf{定量复现：}
\begin{itemize}
\item $\tau\propto 1/\eta$：\hl{$p=0.971$}，$R^2=0.9996$
\item 换种子、留一法均稳健
\item $\beta_2\in[0.9,0.9999]$：$\tau$ 平坦，CV \hl{1.1\%}
\item 幅度恒等：$dL_{\rm eq}/d\eta = 3.075$ vs 闭式 $\frac{1}{4}\sum_i s_i = 3.0$，\hl{误差 $<3\%$}
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{谱混合诊断：}
\begin{itemize}
\item 双峰谱上双指数核提升 $R^2$ 0.065
\item 恢复两个注入速率 $\{1, 16\}$（拟合得 $\{0.87, 16.1\}$）
\item \hl{机制在已知谱上被证实为真正的谱混合}
\end{itemize}
\vspace{0.3em}
\textbf{可证伪性：} 信号主导谱\hl{破坏} $p=1$。
\end{column}

\begin{frame}{诚实负结果与开放问题}
\textbf{在 $\sim$10M transformer 上\emph{未}复现的：}
\begin{itemize}
\item \hl{$\tau\propto 1/\eta$ 在此尺度不成立}：$16\times$ LR 范围上测得 $\tau\approx 475$--$912$ 步（几乎平坦，$p\approx 0.18$）。审计（合成恢复测试 + profile-likelihood）明确排除窗口截断伪影——这是真实测量，与论文"信号随尺度增长"的声明一致。
\item \hl{单曲线核简并}：有限-$\lambda_{\rm slow}$ 核与绝热基线 corr$\approx 0.99$；速率依赖只在\emph{跨调度}对比中可辨识。
\end{itemize}
\vfill
\textbf{可复现的不一致性：}
\begin{itemize}
\item \hl{$\eta_{\rm peak}$ 归一化歧义}：若保留 Eq.(4) 中的 $\eta_{\rm peak}$ 因子，拟合得 $c\approx 60$--$86$；只有丢弃它才得 $c\approx 0.5$。论文的 $c\approx 0.5$ 绑定于特定（未声明的）归一化约定。
\end{itemize}
\vfill
\textbf{开放问题仍开放：}
\begin{itemize}
\item 冻结参数的跨\emph{形状}迁移失败（平均 $R^2=0.52$，未达预注册门槛 0.6）
\item 无 $O(1)$ 拟合常数地从第一性原理计算 $\lambda_{\rm slow}$ 和 $c$ —— 论文开放问题 (a) 未闭合
\end{itemize}
\end{frame}

\section{结论}
\begin{frame}{结论}
\textbf{核心贡献。}
\begin{enumerate}
\item 指出现有最优损失曲线律（MPL/FSL）皆为\hl{绝热近似}——预测准静态平衡值，忽略快速衰减时的弛豫滞后。
\item 从 AdamW 预条件二阶矩动力学\hl{推导}出非绝热修正 \Drop，其幅度等于独立可测的平衡地板敏感度 $dL_{\rm eq}/d\eta$（在受控 NQM 中验证到 3\% 内）。
\item 推出并\hl{实测确认}经典弛豫律 $\tau\propto 1/\eta$（$p=1.00\pm 0.18$），自洽验证噪声主导假设。
\item 独立复现（真实 transformer + NQM 模拟）从两端确认机制，同时\hl{诚实标出边界}（$\tau$ 在 $\sim$10M 未复现、单曲线简并、$\eta_{\rm peak}$ 歧义）。
\item 形状常数在公开同架构尺度上经验稳定 $\Rightarrow$ \hl{低校准跨尺度迁移}，陡降误差 $-$44\%、6/6 全胜。
\item \hl{B-恒等式}：MPL 自带的 $B$ 即地板敏感度（$5$--$15\%$ 内验证）$\Rightarrow$ 零额外测量迁移 $-43\%$。
\item \hl{二阶幅度闭合} $\chi(\eta)\propto\eta^{\delta}$：审计矩阵全指标占优；真实模型界外调度上探针校准 $-18\%\to-71\%$；核形状与速率统一两条死路诚实关闭。
\end{enumerate}
\vfill
\textbf{局限（诚实版）。} 增益限于数据稀缺的陡降；$\lambda_{\rm slow}, c$ 是测量值而非第一性原理计算；$\eta_{\rm peak}$ 归一化有歧义；$\tau\propto 1/\eta$ 在 $\sim$10M 未复现；跨形状迁移失败；单曲线上核简并；所有真实数据仅来自单一公开数据集的三个尺度。
\end{frame}

\begin{frame}{下一步工作}
\textbf{本轮已完成（原优先级 1/2 中的三项）：}
\begin{itemize}
\item \sout{真实模型地板独立测量} → 等-$S$ 阶梯（6 条 constant 段、统一累计 LR）直接测 $F_{\rm eq}(\eta)$ 幂律。
\item \sout{多段递减探针} → \texttt{twodrop} 叠加分解：低 $\eta$ 降的可见滞后 $A_2/A_1\le0.13$，比一切闭合预测更强的抑制。
\item \sout{warmup 对称性} → \texttt{cyclic} 再升温：MPL 骨干自身先失效（残差 $+0.07$），滞后项无法在其上检验。
\end{itemize}
\vfill
\textbf{仍开放：}
\begin{itemize}
\item \hl{S-time 操控实验}：改 batch size 使 step-time profile 相同但 S-time 不同，直接检验记忆变量是 $S$ 而非 $t$。
\item NQM 中换 SGD，检验效应是否不限于 AdamW。
\item $c$ 的浓度依赖（瞬时降 $c\approx0.1$--$0.26$ vs 渐进降 $0.43$--$0.60$）的第一性原理解释。
\item 识别性门槛的跨套件可移植性（invsqrt 过矫案例）。
\end{itemize}

\begin{itemize}
\item 在 NQM 中强制 edge-of-stability 区域，检验是否能产生 MPL 经验指数 $\gamma$ 的行为。
\end{itemize}
