\section{Project Goal}

\begin{frame}{The problem: predict the whole loss curve before training}
\textbf{Goal.} Given a learning-rate schedule $\{\eta_t\}$, predict the pretraining loss
curve $L(t)$ without running the full training job.
\vfill
\textbf{Why this matters.}
\begin{itemize}
\item LLM pretraining is expensive; schedule search by full trial runs is wasteful.
\item Standard scaling laws mostly target endpoint loss; schedule-aware laws must predict
the \emph{trajectory} under warmup, stable phases, cooldowns, and WSD-style drops.
\item The hardest test is cross-schedule transfer: fit on slow schedules such as cosine,
then predict unseen sharp-decay WSD curves.
\end{itemize}
\vfill
\textbf{Project question.}
\begin{center}
\hl{Do current loss-curve laws only know how much LR was spent, or also how fast it was spent?}
\end{center}
\end{frame}

\begin{frame}{Baseline laws: strong, but structurally adiabatic}
Let $S(t)=\sum_{i\le t}\eta_i$ be cumulative LR and
$\Delta\eta_k=\eta_{k-1}-\eta_k$ be a LR decrement.
\vfill
\textbf{Multi-Power Law (MPL).}
\[
L(t)=L_0+A\,S(t)^{-\alpha}
+B\sum_{k\le t}\Delta\eta_k\,
G\!\left(\eta_k^{-\gamma}(S(t)-S(k))\right),
\quad
G(x)=1-(1+Cx)^{-\beta}.
\]
\textbf{Functional Scaling Law (FSL).}
FSL gives a closely related functional predictor over the LR history.
\vfill
\textbf{What they capture well.}
\begin{itemize}
\item Integrated progress through $S(t)$.
\item A saturating benefit from past LR drops.
\item Cross-schedule structure in public LLM curves.
\end{itemize}
\textbf{The gap.} Their response to a drop is essentially quasi-static: after fitting the
schedule kernel, they do not explicitly model the finite-time tracking error of a moving
LR-dependent noise floor. I call this an \hl{adiabatic approximation}.
\end{frame}

\begin{frame}{Core thesis}
\begin{center}
\Large
\hl{Loss depends not only on how much LR has been spent,\\
but also on how fast it was spent.}
\end{center}
\vfill
\textbf{Mechanism.}
\begin{itemize}
\item At a fixed LR, the optimizer has an equilibrium excess-loss/noise-floor state.
\item A sharp LR drop lowers that equilibrium immediately.
\item The actual excess-loss state relaxes only over a finite time.
\item During this transient, the true loss is \hl{higher} than the adiabatic prediction.
\end{itemize}
\vfill
\textbf{Resulting signature.}
Slow and fast schedules may reach the same final LR, but the fast sweep retains a positive
residual. This is not just a function of cumulative LR $S$; it is a rate-dependent lag.
\end{frame}

\begin{frame}{Data source and processing pipeline}
\textbf{Public-curve data.}
\begin{itemize}
\item Source: official MPL public loss curves vendored under
\code{external/MultiPowerLaw/loss\_curve\_repo/csv\_\{25,100,400\}}.
\item Scales: $25$M, $100$M, $400$M.
\item Schedules: cosine, constant, WSD, WSD-long-decay, and WSD constant-tail probes.
\end{itemize}
\vfill
\textbf{Processing pipeline.}
\begin{enumerate}
\item Load per-run CSV curves and learning-rate schedules.
\item Align curves by training step; compute cumulative LR
$S(t)=\sum_{i\le t}\eta_i$ and LR decrements
$\Delta\eta_k=(\eta_{k-1}-\eta_k)_+$.
\item Fit baseline laws on selected training curves using log-loss Huber objectives.
\item Evaluate predictions on held-out WSD-family curves using MAE.
\item Compute residuals $L_{\mathrm{true}}-L_{\mathrm{MPL}}$ and fit/test \Drop{} only on
the residual structure.
\end{enumerate}
\vfill
\textbf{Main parameters.} MPL has $(L_0,A,\alpha,B,C,\beta,\gamma)$; \Drop{} adds
$\lambda_{\mathrm{slow}}$, $c$, and independently measured $\chi=dL_{\mathrm{eq}}/d\eta$.
\end{frame}

\textbf{Course-task reproduction split.}
\begin{itemize}

\item Held-out test curves:
\code{wsd\_20000\_24000}, \code{wsdld\_20000\_24000},

\item Evaluation scales: $25$M, $100$M, and $400$M.
\end{itemize}
\vfill
\textbf{Additional public-curve experiments.}
\begin{itemize}
\item Use a faithfully reproduced MPL as the adiabatic baseline.
\item Measure $\tau$ from two-stage \code{wsdcon} constant-tail windows.
\item Use \code{wsd}/\code{wsdld} as held-out sharp-decay targets for low-calibration

\textbf{Why this protocol is fair.}

\section{Method}

\begin{frame}{Theory in one slide: linear response of a moving equilibrium}
Assume an LR-dependent hidden excess-loss state $z_t$ with local equilibrium
$z^\star(S_t,\eta_t)$:
\[
z_{t+1}=\Phi_{S_t,\eta_t}(z_t),\qquad
L_t=L_{\mathrm{ad}}(S_t,\eta_t)+r_t.
\]
Linearizing around the moving equilibrium gives the cooldown-specific residual
\[
r_t \approx \sum_{k<t}K_{k,t}\,(\eta_k-\eta_{k+1})_+.
\]
\vfill
Under the first-order optimizer approximation
$T_t\simeq I-\eta_t A$, products of transition matrices become exponentials in
\hl{cumulative-LR time}:
\[
T_{t-1}\cdots T_{k+1}\simeq
\exp[-A(S_t-S_k)].
\]
\vfill
\begin{block}{Primitive prediction}
The non-adiabatic residual is a causal convolution of LR decrements with a stable memory
kernel in $S$-time, not a free bump fitted after the fact.
\end{block}
\end{frame}

\begin{frame}{Operational correction: DropRelaxS}
The full kernel can be a spectral mixture. To avoid overfitting, the paper uses the
lowest-capacity one-pole closure:
\[
\boxed{
L(t)=L_{\mathrm{MPL}}(t)+c\,\chi
\sum_{k<t}e^{-\lambda_{\mathrm{slow}}(S_t-S_k)}(\eta_k-\eta_{k+1})_+
}
\]
\[
\chi=\frac{dL_{\mathrm{eq}}}{d\eta}.
\]
\vfill
\textbf{Interpretation.}
\begin{itemize}
\item $(\eta_k-\eta_{k+1})_+$: only LR drops create the positive lag.
\item $e^{-\lambda_{\mathrm{slow}}(S_t-S_k)}$: memory decays in cumulative LR time.
\item $\chi$: amplitude equals the equilibrium noise-floor sensitivity.
\item $c$: low-capacity compression of a spectral mixture into one pole.
\end{itemize}
\vfill
\textbf{Important convention.} If drops are normalized by $\eta_{\mathrm{peak}}$, the
prefactor must absorb that factor. With raw LR drops, the dimensionally clean amplitude is
$c\chi$.
\end{frame}

\begin{frame}{AdamW specialization: expected relaxation scaling}
For a noisy quadratic mode with curvature $\lambda_i$ and gradient-noise standard deviation
$s_i$, noise-dominated AdamW has $v_i^\star\simeq s_i^2$. Locally it behaves like
preconditioned SGD:
\[
x_{i,t+1}=\left(1-\eta_t\frac{\lambda_i}{s_i}\right)x_{i,t}
-\eta_t\frac{1}{s_i}\xi_{i,t}.
\]
For $V_i=\mathbb E[x_i^2]$:
\[
V_i^\star(\eta)=\frac{\eta s_i}{2\lambda_i}+O(\eta^2),\qquad
F_i^\star(\eta)=\frac12\lambda_iV_i^\star(\eta)=\frac{\eta s_i}{4}+O(\eta^2).
\]
\vfill
\textbf{Two useful consistency checks.}
\begin{itemize}
\item \textbf{Amplitude identity:}
$\sum_i dF_i^\star/d\eta=\frac14\sum_i s_i+O(\eta)$.
\item \textbf{Classical relaxation time:}
\[
\tau_i^{-1}=2\eta\frac{\lambda_i}{s_i},
\qquad \tau_i\propto \eta^{-1}.
\]
\end{itemize}
This timescale is not a new theorem; here it is used as a mechanistic check for whether the
\hl{residual left by adiabatic loss-curve laws} behaves like finite-time relaxation.
\end{frame}

\begin{frame}{Predictions and falsifiability}
\textbf{The correction predicts six observable signatures.}
\begin{enumerate}
\item Residual after fast LR drops is positive.
\item Residual depends causally on LR decrements, not just on LR level.
\item Memory is in cumulative LR time $S$, not raw step time.
\item Amplitude is controlled by $dL_{\mathrm{eq}}/d\eta$.
\item In a noise-dominated AdamW regime, residual relaxation should be consistent with
$\tau\propto1/\eta$.
\item A spectral mixture should sometimes beat a single exponential.
\end{enumerate}
\vfill
\textbf{Failure modes are also meaningful.}
\begin{itemize}
\item If a full set of WSD curves is available, MPL can absorb the residual by refitting.
\item If the scale is too small/noisy, this relaxation scaling may not be resolved.
\item A single decay curve can make finite-$\lambda_{\mathrm{slow}}$ nearly degenerate with
the adiabatic limit; rate dependence must be identified across schedules.
\end{itemize}
\end{frame}

\begin{frame}{Upgrade 1: the B-identity --- the amplitude was inside MPL all along}
\textbf{Proposition.} Compare holding $\eta$ vs dropping to $\eta-d\eta$ and equilibrating at
matched $S$: MPL's annealing term changes by $-B\,d\eta\,G(\cdot)\to-B\,d\eta$ as $G\to1$, so
\[
\left.\partial_\eta L^{\mathrm{MPL}}_{\mathrm{eq}}\right|_{S}=B .
\]
\vfill
\textbf{Falsifiable check.} $B=363.8/437.9/523.4$ vs independently measured
$dL_{\mathrm{eq}}/d\eta=381.7/499.3/561.5$ ($5$--$15\%$ at all scales).
\vfill
\begin{block}{Zero-extra-measurement transfer}
$\kappa=c''\eta_{\mathrm{peak}}B$ ($c''$ leave-one-scale-out, CV $14.7\%$) gives
$\mathbf{-43\%}$ ($6/6$) on the Table-1 protocol --- matching the probe chain at zero
additional training runs. Smooth-only MPL refits ($B$ drifts $\le30\%$) retain
$-36$ to $-40\%$.
\end{block}
\end{frame}

\begin{frame}{Upgrade 2: second-order amplitude closure $\chi(\eta)\propto\eta^{\delta}$}
\textbf{Measurement.} Public settled-probe floors read superlinear:
$F_{\mathrm{eq}}\propto\eta^{1+\delta}$, $1{+}\delta=1.06/1.49/1.25$ ($25/100/400$M)
\emph{(protocol-dependent: confound-free equal-$S$ ladders read sublinear $\approx0.65$;
see backup)}. The first-order closure froze $\chi$; the
derivation's forcing $\partial_\eta z^\star(S_k,\eta_k)$ already depends on $\eta_k$.
\[
L(t)=L_{\mathrm{MPL}}(t)+\kappa\sum_{k<t}\Big(\tfrac{\eta_k}{\eta_{\mathrm{peak}}}\Big)^{\!\delta}
e^{-\lambda_{\mathrm{slow}}(S_t-S_k)}\tfrac{(\eta_{k-1}-\eta_k)_+}{\eta_{\mathrm{peak}}}
\]
\vfill
\begin{itemize}\itemsep2pt
\item Public curves, $\delta{=}1/4$ (Pareto-selected, never target-fit): audited
$6{\times}6$ matrix improves on \emph{every} aggregate; probes-only calibration
$-17\%\to-23$ to $-29\%$.
\item Real $10.7$M model, never-scanned out-of-family beds: \texttt{sharp600}
$-18\%\to-71\%$ ($\delta{=}3/4$); showcase \texttt{wsd\_sharp} $-14\%\to-52\%$.
\item Honest split: part of the cheap-probe gain is amplitude recalibration; at matched
amplitude the unweighted shape stays best in-family. Naive NQM route falsified at
magnitude; bulk$+$edge spectrum (tracking cutoff $a_{\max}\!\propto\!1/\eta$) is the
consistent mechanism (conjecture flagged).
\end{itemize}
\end{frame}

\begin{frame}{Fair-baseline diagnostic: formula gap vs information gap}
\textbf{Key question.} Is \Drop{} a better formula, or does it just supply information missing
from a few-shot MPL fit?
\vfill
\textbf{Diagnostic.}
\begin{itemize}
\item If MPL is refit on several full WSD curves, it can absorb the non-adiabatic residual.
\item In that data-rich setting, the explicit \Drop{} term adds little.
\item In the data-poor setting, \Drop{} provides a structured prior that transfers from cheap
probes and other scales.
\end{itemize}
\vfill
\begin{block}{Honest scope}
The correction is an \hl{information-efficient leading-order term}. It is not meant to beat
a fully refit, data-rich MPL on its own training distribution.
\end{block}
\end{frame}

\begin{frame}{Final kappa estimator: remove nuisance drift before transfer}
\textbf{Problem with raw amplitude fitting.}
The residual after MPL contains both the schedule-response feature and smooth MPL drift.
Directly fitting $\kappa$ can get the trend direction right but the magnitude wrong.
\vfill
\textbf{Final calibration model.}
\[
r=L_{\mathrm{obs}}-L_{\mathrm{MPL}}
=\kappa_\star\phi+g_\star+\varepsilon,\qquad g_\star\in\mathcal G,\quad \kappa_\star\ge0.
\]
Let $M_{\mathcal G}$ residualize against the low-frequency nuisance subspace:
\[
\phi_\perp=M_{\mathcal G}\phi,\qquad r_\perp=M_{\mathcal G}r,\qquad
R=\frac{\|\phi_\perp\|_2^2}{\|\phi\|_2^2}.
\]
\vfill
\textbf{Cap-free paper-facing estimator.}
\[
\widehat\kappa
=
\sqrt R
\left(
\frac{\langle \phi_\perp,r_\perp\rangle}
     {\|\phi_\perp\|_2^2+\tau^2}
\right)_+,\qquad \tau=\sigma/k_0.
\]
{\footnotesize Interpretation: FWL partial regression + empirical-Bayes ridge +
identifiable-amplitude conversion.}
\end{frame}

\begin{frame}{How to choose the nuisance subspace}
\textbf{The object is low-frequency drift control, not a polynomial fit.}
Use a DCT low-frequency family:
\[
\mathcal G_K=\operatorname{span}\{q_0,\ldots,q_K\},\qquad
R_K=\frac{\|M_{\mathcal G_K}\phi\|_2^2}{\|\phi\|_2^2}.
\]
\textbf{Automatic spectral rule.}
\[
K^\star=\arg\min_{K\ge K_{\min}} |R_K-\rho|.
\]
\vfill
\begin{center}\small
\begin{tabular}{lccc}
\toprule
Choice of $\mathcal G$ & worst offdiag & mean offdiag & cosine$\to$WSD\\
\midrule
fixed DCT $K=4$ & $-1.8\%$ & $-10.0\%$ & $-3.6\%$\\
retention target only & $+304.0\%$ & $+8.4\%$ & $-9.5\%$\\
$K_{\min}=3,\rho=0.35$ & $-1.7\%$ & $-11.2\%$ & $-10.1\%$\\
\bottomrule
\end{tabular}
\end{center}
\vfill
\textbf{Conclusion.} First enforce minimum low-frequency drift control; then use identifiable
feature energy to choose among sufficiently rich nuisance spaces.
\end{frame}

\begin{frame}{Method comparison: what each model can and cannot express}
\footnotesize
\begin{description}
\item[Tissue.] Simple LR-integral baseline with few parameters. It is useful as a
lightweight reference, but has a coarse schedule response and weaker WSD transfer.
\item[MPL.] Strong adiabatic schedule-aware predictor. Public results are
reproducible, but lag is absorbed only when full WSD curves are available.
\item[MPL+\Drop{}.] Adds a causal finite-time lag term with low calibration cost.
It is most useful for data-poor sharp-decay prediction; constants are still measured
and the single-curve degeneracy remains.
\end{description}
\vfill
\textbf{Empirical comparison.}
\begin{itemize}
\item On the course split, MPL reproduces or improves the existing baselines at $25/400$M
and ties Tissue at $100$M.
\item On held-out sharp decays, low-calibration MPL+\Drop{} reduces MAE by $44\%$ over
well-fit MPL in the public same-architecture setting.
\item When full WSD curves are available, refit MPL absorbs much of the residual; the gain
of \Drop{} is specifically in the ``predict before training'' regime.
\item Exhaustive split audits show the current law can over-correct \code{wsdcon} tails when
calibrated only on sharp decays, so its scope is sharp-cooldown prediction, not arbitrary
residual smoothing.
\end{itemize}
\end{frame}

\textbf{Problem.} The original project data are not fully recoverable: the author-side public

\textbf{Reproduction strategy.}

\begin{frame}{What does not transfer at $\sim$10M}
\textbf{Negative results are part of the final claim.}
\begin{itemize}
\item The real $\sim$10M transformer does \hl{not} resolve $\tau\propto1/\eta$:
over a $16\times$ LR range, measured post-step relaxation is roughly flat
($\tau\approx475$--$912$ steps, $p\approx0.18$).
\item Audits reject the simple ``window truncation'' excuse: synthetic recovery tests and
profile-likelihood checks reject very large hidden $\tau$ values.
\item On a single decay curve, finite-$\lambda_{\mathrm{slow}}$ \Drop{} is numerically close
to the adiabatic limit (correlation $\approx0.99$). Rate dependence is identifiable only
across schedules.
\item Frozen cross-shape transfer in controlled simulations fails its pre-registered
threshold (mean $R^2=0.52<0.6$).
\end{itemize}
\vfill
\begin{block}{Bounded conclusion}
The qualitative non-adiabatic lag is robust. The inverse-LR relaxation trend is useful
supporting evidence in public larger curves and controlled AdamW, but it is not established
by the small real transformer.
\end{block}
\end{frame}

\section{Synthesis}

\begin{frame}{What is solid}
\textbf{Main evidence.}
\begin{itemize}
\item Fast LR decays create a positive residual relative to well-fit adiabatic laws.
\item The residual has the predicted causal LR-decrement structure.
\item Public LLM curves show residual relaxation consistent with inverse-LR scaling
($p=1.00\pm0.18$), as a mechanism check rather than the main claim.
\item Controlled AdamW reproduces the same trend, $\beta_2$ robustness, amplitude identity,
and spectral-mixture behavior.
\item A real $\sim$10M transformer shows no-fitting rate dependence: the fast sweep ends
higher despite more cumulative LR.
\end{itemize}
\vfill
\textbf{Practical value.}
In data-poor sharp-decay prediction, low-calibration \Drop{} transfer reduces held-out MAE by
$44\%$ on public curves (6/6 wins).
\end{frame}

\begin{frame}{What remains open}
\textbf{Constants are measured, not solved.}
\begin{itemize}
\item $\lambda_{\mathrm{slow}}$ is best interpreted as a slow preconditioned mode
$\sim2(\lambda/s)_{\mathrm{eff}}$, not a universal spectral average.
\item A robust first-principles predictor still needs an unexplained $O(1)$ constant.
\item The normalization of LR drops must be stated explicitly; otherwise $c\approx0.5$ and
$c\approx60$--$86$ can appear contradictory while describing the same data under different
conventions.
\end{itemize}
\vfill
\textbf{Scope limits.}
\begin{itemize}
\item Real-data public curves cover one architecture family and three scales.
\item The $\sim$10M independently trained model confirms the effect but not a clean inverse-LR
relaxation trend.
\item The relationship to MPL's empirical exponent $\gamma$ is complementary and still only
qualitative.
\end{itemize}
\vfill
\textbf{Closed by the second (adversarially-audited) round.}
\begin{itemize}
\item The slow-mode \emph{spectral} explanation was tested directly (top-Lanczos
preconditioned sharpness at 10.7M and 25M) and \textbf{falsified} (OFF\_REGIME at both):
$\lambda_{\mathrm{slow}}$ stays phenomenological, not a deterministic edge-re-pinning rate.
\item The floor superlinearity is \textbf{protocol-dependent}: confound-free equal-$S$
ladders read sublinear ($\approx0.65$, robust across $10.7$/$25$M, batch size, horizon to
$24$k); the public $1.06$--$1.49$ is a settled-probe three-point-fit artifact, not a
parameter-count law.
\end{itemize}
\end{frame}

\begin{frame}{Second round: what we ran (all pre-registered, audited)}
\textbf{Priorities 1--3 from the original plan are now DONE.}
\begin{itemize}
\item \emph{Amplitude identity (was P1):} the B-identity ($dL_{\mathrm{eq}}/d\eta=B$) transfers
MPL's own annealing coefficient with no extra measurement ($-43\%$); a split-free ladder
anchor gives $B_l/B_m=1.10$.
\item \emph{S-time / batch control (was P2):} the slow-mode clock is batch-size-\textbf{blind}
(bladder ladder, AUDIT-A C1/C3); the floor exponent is invariant to gradient-noise scale
(bs192 ladder, BS\_NULL $p=0.61$).
\item \emph{Scale + spectrum (was P3):} the $25$M equal-$S$ floor ladder ran ($p=0.64$,
sublinear); top-Lanczos sharpness spectroscopy at $10.7$/$25$M returned OFF\_REGIME --
the spectral slow-mode account is falsified, not open.
\item \emph{Concentration-dependent rate $\Lambda(\rho)$ (now done):} on a wide width
ladder the rate is a real $12\times$ \emph{log-linear}-in-$\log W$ concentration effect,
\emph{not} a fractional/scale-free memory ($\Delta$AIC $22$); promoting to a width-dependent
$\lambda(W)$ buys only $5.9\%$ held-out gain $\Rightarrow$ the fixed effective
$\lambda_{\mathrm{slow}}$ is \textbf{vindicated}. The visible amplitude mirrors it (rises
with $\rho$, Spearman $+0.90$, but mild $2.2\times$, no clean gate).
\end{itemize}
\vfill
\textbf{Genuinely open / next (need a bigger machine).}
\begin{itemize}
\item \emph{P0 (cheap):} more two-stage probe LRs for tighter $\chi/\lambda$ CIs; cross-family
$c$ transfer.
\item \emph{Saturating (Hill) visible-amplitude} for the strong mid-$\eta$ probe deficit:
needs a depth-varied ladder at public $100$/$400$M (where the floor is superlinear) to move
the $\chi$ claim --- $\sim$200--400 GPU-h, beyond one accelerator.
\end{itemize}
\end{frame}

\begin{frame}{Final takeaway}
\begin{center}
\Large
\hl{Learning-rate schedules are not adiabatic.}
\end{center}
\vfill
\textbf{Substantive conclusion.}
MPL/FSL-style laws capture strong quasi-static structure, but fast LR decay leaves a
finite-time relaxation lag. The leading correction is a causal convolution of LR drops with
a memory kernel in cumulative LR time.
\vfill
\textbf{Practical conclusion.}
\Drop{} is useful as a low-calibration prior when full WSD curves are unavailable, especially
for sharp-decay extrapolation.
\vfill
\textbf{Scientific boundary.}
The mechanism is real; the exact constants are not yet first-principles solved. The honest
open problem is to compute $\lambda_{\mathrm{slow}}$ and $c$ from spectra, and to test the
relaxation trend on larger independently trained models.
\end{frame}
