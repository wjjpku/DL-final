\section{Title}

{\setbeamertemplate{footline}{}
\begin{frame}
\titlepage
\vspace{-1.2em}
\footnotesize
\begin{block}{Central question}
Given a cosine training loss curve and a target WSD-family schedule,
can we predict the full WSD loss curve without reading the target WSD loss?
\end{block}
\vspace{-0.45em}
\begin{block}{Thesis}
Cosine residuals contain both a \hl{transferable LR-drop response}
and a \hl{non-transferable MPL-LD parameter error}.
The contribution is to identify these two components and transfer only the former.
\end{block}
\end{frame}}

\begin{frame}{How to read this report}
\small
This report does not start from ``adding a kappa correction''.
It starts from a residual identification problem:
\[
\text{MPL residual transfer is hard because residuals are confounded.}
\]
\vfill
\begin{center}
\scriptsize
\begin{tabular}{L{0.17\linewidth}L{0.51\linewidth}L{0.23\linewidth}}
\toprule
Link & Question & Evidence\\
\midrule

\addlinespace
Obstacle & Can cosine residual be transferred directly? & Direct-kappa negative control.\\
\addlinespace
Identification & How do we remove nuisance drift? & MPL-LD projection.\\
\addlinespace

\section{Problem and obstacle}

\begin{frame}{Problem setting: source cosine \(\rightarrow\) target WSD}
\small
\textbf{Inputs.}
\begin{itemize}\tightlist
\item source: a cosine learning-rate schedule and its observed training loss curve;
\item baseline: frozen MPL predictions for both source and target schedules;
\item target: a WSD-family learning-rate schedule \(\{\eta_t\}_{t=1}^T\).
\end{itemize}
\vfill
\textbf{Output.}
\[
\widehat L_s(t),\qquad t=1,\ldots,T.
\]
\vfill
\textbf{Restriction.}
\[
\text{calibration uses cosine residual only,}
\qquad
\text{target loss is used only for evaluation.}
\]
\vfill
\textbf{Why this is not ordinary curve fitting.}
Cosine and WSD schedules have different decay geometry.
If a source residual component transfers to WSD, it must correspond to a schedule-response mechanism,
not just a low-frequency fitting error on the source curve.
\end{frame}

\footnotesize
\textbf{Model scales.}
\[
25\mathrm{M},\qquad 100\mathrm{M},\qquad 400\mathrm{M}.
\]
\vfill
\textbf{Source curve.}
\[
\code{cosine\_72000.csv},\qquad

\]
The residual calibration uses the source suffix \(t\ge8000\);

\begin{frame}{Data processing pipeline and fixed protocol}
\small
\begin{columns}[T]
\begin{column}{0.58\textwidth}
\textbf{Data processing pipeline}
\begin{enumerate}\tightlist
\item Read source cosine loss and LR schedule.
\item Read frozen MPL prediction.
\item Compute source residual:
\[
r_{\cos}(t)=L_{\cos}(t)-L_{\mathrm{MPL},\cos}(t).
\]
\item Use calibration suffix \(t\ge8000\).
\item For each target WSD schedule, compute LR drops \(d_t\) and response feature \(\phi\).
\item Construct MPL-LD tangent basis and projection \(P_{\mathrm{LD}}\).
\item Estimate source-only \(\widehat\kappa\).
\item Predict target loss; target loss is used only after prediction for evaluation.
\end{enumerate}
\end{column}
\begin{column}{0.38\textwidth}
\textbf{Fixed protocol}
\begin{itemize}\tightlist
\item source suffix: \(t\ge8000\)
\item logging interval: \(\Delta_{\mathrm{obs}}=128\)
\item response time: \(q_2\) half-life rule
\item ridge floor: \(1/N_{\mathrm{cal}}\)
\item constraint: \(\kappa\ge0\)
\item metrics: MAE, RMSE, terminal error
\end{itemize}
\vfill
\textbf{Data source.}
All main loss curves are stored under
\code{external/MultiPowerLaw/...}, with subfolders
\code{csv\_25}, \code{csv\_100}, and \code{csv\_400}.
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Key obstacle: cosine residual is a mixed signal}
\small
If we directly fit a response amplitude from the cosine residual,
\[
\widehat\kappa_{\mathrm{direct}}
=
\frac{\langle \phi_{\lambda,\cos},r_{\cos}\rangle_+}
{\|\phi_{\lambda,\cos}\|_2^2+1/N_{\mathrm{cal}}},
\]
we implicitly assume:
\[
r_{\cos}(t)\approx \kappa\phi_{\lambda,\cos}(t).
\]
This assumption is too strong.
\vfill
A more realistic local decomposition is:
\[
\boxed{
r_{\cos}(t)
=
\underbrace{\kappa\phi_{\lambda,\cos}(t)}_{\text{transferable LR-drop response}}
+
\underbrace{J_{\mathrm{LD}}(t)\Delta\theta_{\mathrm{LD}}}_{\text{non-transferable MPL-LD nuisance}}
+
\epsilon(t)
}
\]
\vfill
\textbf{Intuition.}
Cosine decay is smooth, so \(\phi_{\lambda,\cos}\) is low-frequency.
MPL-LD parameter error also creates low-frequency drift.
Without separating them, \(\kappa\) absorbs non-transferable MPL parameter error.
\end{frame}

\begin{block}{Task reformulation}
Separate the following two components in MPL residuals:
\[
\text{transferable LR-drop response}
\quad\text{vs.}\quad
\text{MPL-LD nuisance drift}.
\]
Transfer only the former.
\end{block}
\end{frame}

\section{Identification method}

\begin{frame}{Core idea: project out MPL-LD tangent nuisance}
\small
The MPL learning-rate dependent term depends on:
\[
\theta_{\mathrm{LD}}=(B,C,\beta,\gamma).
\]
If these parameters have small errors, the first-order MPL prediction error is:
\[
\Delta L_{\mathrm{MPL}}(t)
\approx
J_{\mathrm{LD}}(t)\Delta\theta_{\mathrm{LD}},
\]
where:
\[
J_{\mathrm{LD}}(t)
=
\left[
\frac{\partial L_{\mathrm{MPL}}}{\partial\log B},
\frac{\partial L_{\mathrm{MPL}}}{\partial\log C},
\frac{\partial L_{\mathrm{MPL}}}{\partial\log\beta},
\frac{\partial L_{\mathrm{MPL}}}{\partial\log\gamma}
\right].
\]
\vfill
Let \(P_{\mathrm{LD}}\) be the orthogonal projection onto the span of \(J_{\mathrm{LD}}\).
Before estimating response amplitude, use:
\[
x=(I-P_{\mathrm{LD}})\phi_{\lambda,\cos},
\qquad
y=(I-P_{\mathrm{LD}})r_{\cos}.
\]
\vfill
\textbf{Meaning.}
Estimate the transferable response only in residual directions that local MPL-LD parameter perturbations cannot explain.
\end{frame}

\begin{frame}{What projection removes from cosine residual}
\centering
\includegraphics[width=0.97\linewidth]{fig_projection_decomposition_cosine_100M.png}
\vfill
\raggedright
\small
\textbf{Reading the figure.}
The left panel shows that raw cosine residual and raw response feature are both low-frequency.
The middle panel shows that the MPL-LD tangent space explains much of the smooth residual drift.
The right panel shows that after nuisance removal,
the estimated \(\kappa\) comes from the alignment between projected residual and projected feature.
\vfill
\textbf{Conclusion.}
Projection is not a decorative linear-algebra step.
It removes non-transferable MPL-LD drift from the source cosine residual.
\end{frame}

\begin{frame}{Prediction rule: minimal deployable model}
\small
After projected identification, WSD prediction uses:
\[
\boxed{
\widehat L_s(t)
=
L_{\mathrm{MPL},s}(t)
+
\widehat\kappa_s\phi_{\lambda_s,s}(t)
}
\]
\vfill
\begin{center}
\begin{tabular}{L{0.23\linewidth}L{0.62\linewidth}}
\toprule
Part & Meaning\\
\midrule
\(L_{\mathrm{MPL},s}(t)\) & frozen MPL baseline.\\
\(\phi_{\lambda_s,s}(t)\) & causal LR-drop response shape induced by the target schedule.\\
\(\lambda_s\) & response decay rate fixed by the target schedule using the \(q_2\) half-life rule.\\
\(\widehat\kappa_s\) & the only scalar estimated from loss residual; it uses source cosine residual only.\\
\bottomrule
\end{tabular}
\end{center}
\vfill
\textbf{One sentence.}
The target schedule determines the response shape;
the source residual determines the response amplitude.
WSD prediction does not read WSD loss.
\end{frame}

\begin{frame}{Causal LR-drop response: schedule-only shape}
\small
Define positive LR drop:
\[
d_t=[\eta_{t-1}-\eta_t]_+.
\]
When LR does not decrease, \(d_t=0\).
\vfill
Define the response recursion:
\[
z_t=\exp(-\lambda_s\eta_t)z_{t-1}+d_t,
\qquad
\phi_{\lambda_s,s}(t)=\frac{z_t}{\eta_{\max}}.
\]
Equivalently:
\[
\phi_{\lambda_s,s}(t)
=
\frac{1}{\eta_{\max}}
\sum_{u\le t}
d_u
\exp\left(
-\lambda_s\sum_{v=u+1}^{t}\eta_v
\right).
\]
\vfill
\textbf{Interpretation.}
An LR drop is a forcing event.
Each drop injects a residual impulse, and the impulse decays in cumulative LR time.
Therefore \(\phi\) is schedule-only, causal, and target-loss-free.
\vfill
\textbf{Response time scale.}
\(\lambda_s\) is fixed by a schedule-only half-life rule:
diffuse LR decay receives a longer response time,
while a sharp LR drop receives a shorter response time.
\end{frame}

\begin{frame}{Source-only one-dimensional \(\widehat\kappa\)}
\small
On the source cosine suffix, define:
\[
x_s=(I-P_{\mathrm{LD}})\phi_{\lambda_s,\cos},
\qquad
y=(I-P_{\mathrm{LD}})r_{\cos}.
\]
The estimation problem is:
\[
\min_{\kappa\ge0}
\|y-\kappa x_s\|_2^2+\frac{1}{N_{\mathrm{cal}}}\kappa^2.
\]
Closed-form solution:
\[
\boxed{
\widehat\kappa_s
=
\frac{[x_s^\top y]_+}
{\|x_s\|_2^2+1/N_{\mathrm{cal}}}
}
\]
\vfill
\textbf{Interpretation.}
\(\widehat\kappa_s\) is the amplitude of the projected response feature.
It is not a multi-parameter residual regressor.
If \(x_s\) and \(y\) have no positive alignment after projection,
the response is not activated.
\end{frame}

\begin{frame}{Appendix A: q2 half-life rule}
\small
Loss is observed only on discrete logging points. Let the main logging interval be:
\[
\Delta_{\mathrm{obs}}=128,
\qquad
\lambda_{\mathrm{obs}}
=
\frac{\log 2}{\eta_{\max}\Delta_{\mathrm{obs}}}.
\]
\vfill
Let total drop and normalized drop mass be:
\[
D_s=\sum_t d_t,
\qquad
p_t=\frac{d_t}{D_s}.
\]
Define Herfindahl concentration:
\[
q_s=\sum_t p_t^2=\frac{1}{n_{\mathrm{eff}}}.
\]
This work uses:
\[
\boxed{
\lambda_s=\frac{\lambda_{\mathrm{obs}}}{2-q_s}
}
\]
\vfill
\textbf{Intuition.}
Diffuse decay receives roughly a \(2\Delta_{\mathrm{obs}}\) half-life;
a single sharp drop receives roughly a \(\Delta_{\mathrm{obs}}\) half-life.
The rule reads only the schedule and not target loss.
\end{frame}

\begin{frame}{Appendix B: Why one-pole exponential response?}
\small
This work treats LR drop as forcing and residual relaxation as a low-capacity one-pole closure:
\[
z_t=\exp(-\lambda_s\eta_t)z_{t-1}+d_t.
\]
\vfill
Three points:
\begin{enumerate}\tightlist
\item LR drop is a discrete forcing: \(d_t=[\eta_{t-1}-\eta_t]_+\).
\item If recovery from forcing is approximated by one state variable, the state decays in cumulative LR time.
\item The Green's function of a first-order linear system is an exponential kernel:
\[
\exp\left(-\lambda_s\sum_{v=u+1}^{t}\eta_v\right).
\]
\end{enumerate}
\vfill
\textbf{Boundary.}
This is not a strict training-dynamics theorem.
It is a one-pole closure assumption.
Kernel ablations test whether it is more reasonable than same-capacity alternatives.
\end{frame}
